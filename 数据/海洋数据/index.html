function GetRad,u,v
if(u eq 0) then begin
 if(v gt 0) then begin 
 return, 3.1415926/2.0
 endif else begin
 return, 3.1415926/2.0*3.0
 endelse
 endif else begin
 tn=atan(v/u)
 
 if(u gt 0) then begin
  
 return, tn lt 0?tn+3.1415926*2:tn
 endif else begin
  return, tn+3.1415926
 endelse 
 endelse
 
 tn=atan(v/u)
end
